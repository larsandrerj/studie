  private static List<Tag> GenerateTagsListFromDB(TagService ts, String[] tags) {
    List<Tag> tagsList = new ArrayList<>();
    for (String tag : tags) {
      ts.getObjectFromDatabase(tag).ifPresent(tagsList::add);
    }
    return tagsList;
  }
