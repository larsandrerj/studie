package no.ntnu.idatt1005.model.utils;

import no.ntnu.idatt1005.model.data.IDAmountStruct;

import java.util.List;

public class EntityValidator {
    public static void validateName(String s) {
        if (s == null || s.isBlank() || s.length() >= 40){
            throw new IllegalArgumentException("Name must be between 1 and 40 characters");
        }
    }

    public static void validateDefaultAmount(float n) {
        if (n < 0 || n > 100000) {
            throw new IllegalArgumentException("Default amount must be between 0 and 100'000");
        }
    }

    public static void validatePrice(int n) {
        if (n < 0 || n > 1000) {
            throw new IllegalArgumentException("Price must be between 0 and 1000");
        }
    }

    public static void validateUnitOfMeasurement(String s) {
        if (s == null || s.isBlank() || s.length() > 20) {
            throw new IllegalArgumentException("Unit of measurement must be between 1 and 20 characters");
        }
    }

    public static void validateInstructions(String s) {
        if (s == null || s.isBlank() || s.length() > 10000) {
            throw new IllegalArgumentException("Instructions must be between 1 and 1000 characters");
        }
    }

    public static void validatePortions(int n) {
        if (n < 1 || n > 100) {
            throw new IllegalArgumentException("Portions must be between 1 and 100");
        }
    }

    public static void validateList(List<?> list) {
        if (list == null || list.isEmpty()) {
            throw new IllegalArgumentException("List cannot be empty");
        }
    }
}
